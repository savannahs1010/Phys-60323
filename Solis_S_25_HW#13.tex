\begin{equation*}
\Psi{(x)}=\begin{cases}
			A \frac{x}{a}, & 0 \leq x \leq a\\[8pt]
            A \frac{(b-x)}{(b-a)}, & a \leq x \leq b\\[8pt]
            0, & \text{otherwise}
		 \end{cases}
\end{equation*}

    \item (3.4 points) What is the expectation value of $x$?.
\end{enumerate}
